cmake_minimum_required(VERSION 3.10)
project(ImageClass)
find_package(OpenCV REQUIRED)
find_package(Qt6Core REQUIRED)
find_package(Qt6Widgets REQUIRED)
include_directories(header)
include_directories(${CMAKE_CURRENT_SOURCE_DIR})
include_directories(/usr/local/Cellar/opencv/4.5.1_3/include)
file(GLOB SOURCES "sources/*.cpp")
add_library(source STATIC ${SOURCES} )
target_link_libraries(source Qt6::Widgets Qt6::Core ${OpenCV_LIBS})
add_executable(img ${SOURCES})
target_link_libraries(img source)
